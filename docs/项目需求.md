### 3.2 VFS 接口设计（关键）

通过抽象基类统一两种文件系统，上层代码无需关心底层是 inode 还是 FAT：

```cpp
class IFileSystem {
public:
    virtual int     fs_format()                              = 0;
    virtual int     fs_mount()                               = 0;
    virtual int     fs_unmount()                             = 0;

    virtual int     fs_open(const char* path, int flags)     = 0;
    virtual int     fs_close(int fd)                         = 0;
    virtual ssize_t fs_read(int fd, void* buf, size_t len)   = 0;
    virtual ssize_t fs_write(int fd, const void* buf, size_t len) = 0;

    virtual int     fs_create(const char* path, int mode)    = 0;
    virtual int     fs_delete(const char* path)              = 0;
    virtual int     fs_mkdir(const char* path)               = 0;
    virtual int     fs_rmdir(const char* path)               = 0;
    virtual int     fs_chdir(const char* path)               = 0;
    virtual int     fs_ls(const char* path, std::vector<DirEntry>& out) = 0;

    virtual int     fs_chmod(const char* path, int mode)     = 0;
    virtual int     fs_link(const char* src, const char* dst)= 0;
    virtual int     fs_stat(const char* path, FileStat& out) = 0;

    virtual std::string fs_type_name() const                 = 0;
    virtual DiskUsage   fs_disk_usage() const                = 0;

    virtual ~IFileSystem() = default;
};
```

### 3.3 UNIX FS 磁盘布局

```
块号:  [0]  [1]  [2 .. 33]            [34 .. 545]
       超级块     inode 区 (32块×16个)    数据区 (512块)
                  共 512 个 inode
```

| 区域 | 块数 | 说明 |
|------|------|------|
| 超级块 | 1 | 空闲 inode 栈 + 空闲块栈 + 元信息 |
| 引导块 | 1 | 预留（可存版本信息） |
| inode 区 | 32 | 每 inode 64B，每块 8 个，共 256 个 inode |
| 数据区 | 512 | 成组块链接法管理，每组 50 块 |

inode 混合索引：
- `addr[0..5]`：直接索引 → 3KB
- `addr[6..7]`：一次间址 → 每块存 128 个块号 → +128KB
- `addr[8]`：二次间址 → 128×128 → +8MB
- `addr[9]`：预留（三次间址，可选）

### 3.4 FAT16 磁盘布局

```
块号:  [0]     [1..4]  [5..8]  [9]           [10 .. 545]
       引导块   FAT1    FAT2    根目录区(1块)   数据区
```

| 区域 | 说明 |
|------|------|
| 引导块 | 卷标、块大小、FAT 大小等 |
| FAT1/FAT2 | 文件分配表（双份备份），每项 2 字节 |
| 根目录区 | 固定大小，存放根目录项 |
| 数据区 | 簇链分配 |

### 3.5 关键常量（与课设 PPT 对齐）

```cpp
constexpr int BLOCK_SIZE     = 512;
constexpr int INODE_SIZE     = 64;     // 每个 inode 占 64 字节
constexpr int INODES_PER_BLK = BLOCK_SIZE / INODE_SIZE;  // 8
constexpr int INODE_BLK_NUM  = 32;

constexpr int DATA_BLK_NUM   = 512;
```

## 4 数据结构设计

### 4.1 UNIX FS 核心结构

```cpp
// 磁盘 inode（64 字节，紧凑存储）
struct DiskINode {
    uint16_t i_mode;          // 文件类型 + rwx 权限
    uint16_t i_nlink;         // 硬链接计数
    uint16_t i_uid;           // 所有者 ID
    uint16_t i_gid;           // 组 ID
    uint32_t i_size;          // 文件大小（字节）
    uint32_t i_atime;         // 最后访问时间
    uint32_t i_mtime;         // 最后修改时间
    uint32_t i_ctime;         // 创建时间
    uint32_t i_addr[NADDR];   // 混合索引地址数组
    uint8_t  _pad[4];         // 对齐到 64 字节
};

// 内存 inode（运行时使用）
struct MemINode {
    DiskINode  d_inode;       // 磁盘 inode 副本
    uint16_t   i_ino;         // inode 编号
    uint16_t   i_count;       // 引用计数
    bool       i_dirty;       // 是否被修改
    MemINode*  i_hash_next;   // hash 链表
};

// 超级块
struct SuperBlock {
    uint16_t s_inode_cnt;              // inode 总数
    uint16_t s_free_inode_cnt;         // 空闲 inode 数
    uint16_t s_ninode;                 // 空闲 inode 栈中的数量
    uint16_t s_inode_stack[NICINOD];   // 空闲 inode 栈

    uint16_t s_block_cnt;              // 数据块总数
    uint16_t s_free_block_cnt;         // 空闲数据块数
    uint16_t s_nfree;                  // 空闲块栈中的数量
    uint16_t s_free_stack[NICFREE];    // 空闲块栈（成组链）

    bool     s_dirty;                  // 修改标志
};

// 目录项
struct DirEntry {
    char     name[MAX_FILENAME];  // 文件名 (14B)
    uint16_t inode_no;            // inode 编号 (2B)
};

// 系统打开文件表项
struct SysOpenFile {
    uint16_t f_mode;       // 打开模式 (r/w/rw)
    uint16_t f_count;      // 引用计数
    uint16_t f_inode_no;   // 对应 inode 编号
    uint32_t f_offset;     // 读写偏移
};

// 用户打开文件表（每用户）
struct UserOpenFile {
    int sys_file_idx[MAX_OPEN_FILE];  // 指向系统打开文件表的索引，-1 为空
};
```

### 4.2 FAT16 核心结构

```cpp
// 引导扇区
struct FAT16BootSector {
    char     volume_label[11];
    uint16_t bytes_per_sector;    // 512
    uint8_t  sectors_per_cluster; // 1
    uint16_t reserved_sectors;    // 1
    uint8_t  num_fats;            // 2
    uint16_t root_entry_count;    // 32 (每项 32 字节，占 2 个块)
    uint16_t total_sectors;
    uint16_t fat_size_sectors;    // 每个 FAT 表占的块数
};

// FAT16 目录项（32 字节，兼容 DOS）
struct FAT16DirEntry {
    char     name[8];       // 文件名
    char     ext[3];        // 扩展名
    uint8_t  attr;          // 属性 (目录/文件/只读/隐藏...)
    uint8_t  reserved[10];
    uint16_t time;          // 修改时间
    uint16_t date;          // 修改日期
    uint16_t first_cluster; // 起始簇号
    uint32_t file_size;     // 文件大小
};
```
